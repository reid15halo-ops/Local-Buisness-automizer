SECTION 3: CABLE LABELS  (COLOR CODING LEGEND)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

COLOR CODE:
  BLUE  cables  = Compute devices  (ThinkCentre, Gaming Rig, HP t640)
  RED   cables  = Gateway/Uplink   (Router → Switch, Pi4 as router, WAN)
  GREEN cables  = NAS / Storage    (NAS 2300, backup paths)
  GREY  cables  = Management       (IPMI, serial console, USB)
  WHITE cables  = Phone / Comms    (USB charging, Termux tether)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SECTION 7: P-TOUCH PRINT SETTINGS

  Recommended tape: Brother TZe-231 (12mm, Black on White)
  For color-coded cables: TZe-431 (Black on Red), TZe-531 (Black on Blue),
                          TZe-731 (Black on Green), TZe-221 (Black on White)
  Cable flag labels: Use TZe tape + Flag label template in P-Touch Editor
  Device panel labels: TZe-231 12mm (main) + TZe-221 9mm (sub-label)
  IP address labels: TZe-621 (Black on Yellow) 9mm
  Rack overview card: TZe-335 (White on Black) 12mm for rack ID

  P-Touch Editor template: "Flag" → text both sides, 25mm flag width
  Font: Helsinki Bold 12pt (device labels), Helsinki Regular 8pt (IPs)
